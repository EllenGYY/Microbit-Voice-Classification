113 result.z = z + other.z;
114
115 return result;
116 }
117
118 bool operator==(const Sample3D& other) const
119 {
120 return x == other.x && y == other.y && z == other.z;
121 }
122
123 bool operator!=(const Sample3D& other) const
124 {
125 return !(x == other.x && y == other.y && z == other.z);
126 }

128 float dSquared(Sample3D &s)
129 {
130 float dx = x - s.x;
131 float dy = y - s.y;
132 float dz = z - s.z;
133
134 return (dx*dx) + (dy*dy) + (dz*dz);
135 }
