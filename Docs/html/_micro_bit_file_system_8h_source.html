65//
66// Every file in the file system has a file descriptor.
67// These are held in directory entries, using the following
68// structure.
69//
70struct DirectoryEntry
71{
72 char file_name[MBFS_FILENAME_LENGTH]; // Name of the file.
73 uint16_t first_block; // Logical block address of the start of the file.
74 uint16_t flags; // Status of the file.
75 uint32_t length; // Length of the file in bytes.
76};
77
78//
79// A directory is a list of DirectoryEntry structures:
80//
81struct Directory
82{
83 DirectoryEntry entry[0];
84};
85
86//
87// A FileDescriptor holds contextual information needed for each OPEN file.
88//
89struct FileDescriptor
90{
91 // read/write/creat flags.
92 uint16_t flags;
93
94 // FileDescriptor id
95 uint16_t id;
96
97 // current file position, in bytes.
98 uint32_t seek;
99
100 // the current file size. n.b. this may be different to that stored in the DirectoryEntry.
101 uint32_t length;
102
103 // the directory entry of this file.
104 DirectoryEntry *dirent;
105
106 // the directory entry of our parent directory.
107 DirectoryEntry *directory;
108
109 // We maintain a chain of open file descriptors. Reference to the next FileDescriptor in the chain.
110 FileDescriptor *next;
111
112 // Optional writeback cache, to minimise FLASH write operations at the expense of RAM.
113 uint16_t cacheLength;
114 uint8_t cache[MBFS_CACHE_SIZE];
115};
