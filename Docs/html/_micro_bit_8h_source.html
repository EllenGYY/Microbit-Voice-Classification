263 inline ManagedString MicroBit::getSerial()
264 {
265 uint64_t n = target_get_serial();
266 int d = 1000000000;
267 int n1 = n % d; n /= d;
268 int n2 = n % d; n /= d;
269 int n3 = n % d; n /= d;
270 ManagedString s1(n1);
271 ManagedString s2(n2);
272 ManagedString s3(n3);
273 ManagedString s = s3 + s2 + s1;
274 return s;
275
276 }
