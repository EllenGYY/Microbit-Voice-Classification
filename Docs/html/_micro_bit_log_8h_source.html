25/*
26 * Log file system format:
27 *
28 * H = strlen(MICROBIT_LOG_HEADER) rounded up to page boundary.
29 * N = (dataStart - H * PAGE_SIZE) / PAGE_SIZE
30 * M = logLength - (N+H)*PAGE_SIZE
31 *
32 * H x Header pages.
33 * +===========================+
34 * | PREEFINED-HEADER |
35 * +===========================+
36 *
37 * +===========================+
38 * | |
39 * | MicroBitLogMetaData |
40 * | (40 bytes) |
41 * | |
42 * +---------------------------+
43 * | |
44 * | 0x00 Init Region |
45 * | (variable length) |
46 * | |
47 * +---------------------------+
48 * | |
49 * | Comma separated Log Keys |
50 * | (variable length) |
51 * | |
52 * +---------------------------|
53 * | |
54 * | 0xFF Init Region |
55 * | (variable length) |
56 * | |
57 * +===========================+
58 *
59 *
60 * N x Journal Pages:
61 * +===========================+
62 * | |
63 * | 0x00 Init Region |
64 * | (variable length) |
65 * | |
66 * +---------------------------+
67 * | |
68 * | Logfile length |
69 * | (12 bytes) |
70 * | |
71 * +---------------------------|
72 * | |
73 * | 0xFF Init Region |
74 * | (variable length) |
75 * | |
76 * +===========================+
77 *
78 * M x pages
79 * +===========================+
80 * | |
81 * | Log Data |
82 * | |
83 * +---------------------------|
84 * | |
85 * | 0xFF Init Region |
86 * | (variable length) |
87 * | |
88 * +===========================+
89 *
90 */

154 struct JournalEntry
155 {
156 char length[MICROBIT_LOG_JOURNAL_ENTRY_SIZE];
157 char null;
158
159 JournalEntry()
160 {
161 memcpy(length, "00000000", MICROBIT_LOG_JOURNAL_ENTRY_SIZE);
162 null = 0;
163 }
164
165 void clear()
166 {
167 memset(length, 0, MICROBIT_LOG_JOURNAL_ENTRY_SIZE);
168 }
169
170 bool containsOnly(uint8_t value)
171 {
172 for (int i=0; i<MICROBIT_LOG_JOURNAL_ENTRY_SIZE;i++)
173 if (length[i] != value)
174 return false;
175
176 return true;
177 }
178 };
