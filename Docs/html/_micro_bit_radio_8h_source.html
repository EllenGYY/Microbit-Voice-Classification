88 struct FrameBuffer
89 {
90 uint8_t length; // The length of the remaining bytes in the packet. includes protocol/version/group fields, excluding the length field itself.
91 uint8_t version; // Protocol version code.
92 uint8_t group; // ID of the group to which this packet belongs.
93 uint8_t protocol; // Inner protocol number c.f. those issued by IANA for IP protocols
94
95 uint8_t payload[MICROBIT_RADIO_MAX_PACKET_SIZE]; // User / higher layer protocol data
96 FrameBuffer *next; // Linkage, to allow this and other protocols to queue packets pending processing.
97 int rssi; // Received signal strength of this frame.
98 };
