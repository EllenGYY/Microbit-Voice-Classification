207template<typename T>
208ManagedType<T>::~ManagedType()
209{
210 // Special case - we were created using a default constructor, and never assigned a value.
211 if (*ref == 0)
212 {
213 // Simply destroy our reference counter and we're done.
214 free(ref);
215 }
216
217 // Normal case - we have a valid piece of data.
218 // Decrement our reference counter and free all allocated memory if we're deleting the last reference.
219 else if (--(*ref) == 0)
220 {
221 delete object;
222 free(ref);
223 }
224}

235template<typename T>
236ManagedType<T>& ManagedType<T>::operator = (const ManagedType<T>&t)
237{
238 if (this == &t)
239 return *this;
240
241 // Special case - we were created using a default constructor, and never assigned a value.
242 if (*ref == 0)
243 {
244 // Simply destroy our reference counter, as we're about to adopt another.
245 free(ref);
246 }
247
248 else if (--(*ref) == 0)
249 {
250 delete object;
251 free(ref);
252 }
253
254 object = t.object;
255 ref = t.ref;
256
257 (*ref)++;
258
259 return *this;
260}
