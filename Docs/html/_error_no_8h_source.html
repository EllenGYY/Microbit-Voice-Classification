34enum ErrorCode{
35
36 // No error occurred.
37 DEVICE_OK = 0,
38
39 // Invalid parameter given.
40 DEVICE_INVALID_PARAMETER = -1001,
41
42 // Requested operation is unsupported.
43 DEVICE_NOT_SUPPORTED = -1002,
44
45 // Device calibration errors
46 DEVICE_CALIBRATION_IN_PROGRESS = -1003,
47 DEVICE_CALIBRATION_REQUIRED = -1004,
48
49 // The requested operation could not be performed as the device has run out of some essential resource (e.g. allocated memory)
50 DEVICE_NO_RESOURCES = -1005,
51
52 // The requested operation could not be performed as some essential resource is busy (e.g. the display)
53 DEVICE_BUSY = -1006,
54
55 // The requested operation was cancelled before it completed.
56 DEVICE_CANCELLED = -1007,
57
58 // I2C Communication error occured (typically I2C module on processor has locked up.)
59 DEVICE_I2C_ERROR = -1010,
60
61 // The serial bus is currently in use by another fiber.
62 DEVICE_SERIAL_IN_USE = -1011,
63
64 // The requested operation had no data to return.
65 DEVICE_NO_DATA = -1012,
66
67 // function is not implemented.
68 DEVICE_NOT_IMPLEMENTED = -1013,
69
70 // SPI Communication error occured
71 DEVICE_SPI_ERROR = -1014,
72
73 // An invalid state was detected (i.e. not initialised)
74 DEVICE_INVALID_STATE = -1015,
75};

80enum PanicCode{
81 // PANIC Codes. These are not return codes, but are terminal conditions.
82 // These induce a panic operation, where all code stops executing, and a panic state is
83 // entered where the panic code is diplayed.
84
85 // Out out memory error. Heap storage was requested, but is not available.
86 DEVICE_OOM = 20,
87
88 // Corruption detected in the codal device heap space
89 DEVICE_HEAP_ERROR = 30,
90
91 // Dereference of a NULL pointer through the ManagedType class,
92 DEVICE_NULL_DEREFERENCE = 40,
93
94 // Non-recoverable error in USB driver
95 DEVICE_USB_ERROR = 50,
96
97 // Non-recoverable error in the JACDAC stack
98 DEVICE_JACDAC_ERROR = 60,
99
100 // CPU SDK faults
101 DEVICE_CPU_SDK = 70,
102
103 // hardware incorrect configuration
104 DEVICE_HARDWARE_CONFIGURATION_ERROR = 90,
105};
