30 // Status Field flags...
31#define IO_STATUS_DIGITAL_IN 0x0001 // Pin is configured as a digital input, with no pull up.
32#define IO_STATUS_DIGITAL_OUT 0x0002 // Pin is configured as a digital output
33#define IO_STATUS_ANALOG_IN 0x0004 // Pin is Analog in
34#define IO_STATUS_ANALOG_OUT 0x0008 // Pin is Analog out
35#define IO_STATUS_TOUCH_IN 0x0010 // Pin is a makey-makey style touch sensor
36#define IO_STATUS_EVENT_ON_EDGE 0x0020 // Pin will generate events on pin change
37#define IO_STATUS_EVENT_PULSE_ON_EDGE 0x0040 // Pin will generate events on pin change
38#define IO_STATUS_INTERRUPT_ON_EDGE 0x0080 // Pin will generate events on pin change
39#define IO_STATUS_ACTIVE_HI 0x0100 // Pin is ACTIVE_HI if set, or ACTIVE_LO if clear
40#define IO_STATUS_WAKE_ON_ACTIVE 0x0200 // Pin should trigger power manager wake-up
